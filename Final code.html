using namespace std;

// Structure to store house information
struct House {
    string id;
    double x, y;
    double pressure = 100.0; // Default pressure for each house
    vector> pipelines; // (connected house, cost)
};

// Function to calculate the Euclidean distance between two houses
double calculateDistance(const House& house1, const House& house2) {
    return sqrt(pow(house2.x - house1.x, 2) + pow(house2.y - house1.y, 2));
}

// Function to adjust the water pressure within safe limits
void adjustPressure(vector& houses) {
    for (auto& house : houses) {
        double totalPipelineCost = 0;
        for (const auto& pipeline : house.pipelines) {
            totalPipelineCost += pipeline.second;
        }
        if (totalPipelineCost > 100.0) {
            house.pressure -= (totalPipelineCost - 100.0) / 10.0;
        } else {
            house.pressure += (100.0 - totalPipelineCost) / 20.0;
        }

        if (house.pressure < 80) {
            house.pressure = 80;
            cout << "Warning: House " << house.id << " had low pressure. Adjusted to 80 units." << endl;
        } else if (house.pressure > 120) {
            house.pressure = 120;
            cout << "Warning: House " << house.id << " had excessive pressure. Adjusted to 120 units." << endl;
        }
    }
}

// Function to display the pipeline network
void displayNetwork(const vector& houses) {
    cout << "Pipeline Network:" << endl;
    for (const auto& house : houses) {
        cout << "House ID: " << house.id << " Pressure: " << house.pressure << " units" << endl;
        for (const auto& pipeline : house.pipelines) {
            cout << "  - Connected to " << pipeline.first << " with cost: " << pipeline.second << endl;
        }
    }
}

// Main function
int main() {
    vector houses;

    int n;
    cout << "Enter the number of houses: ";
    cin >> n;

    for (int i = 0; i < n; ++i) {
        House house;
        cout << "Enter details for house " << i + 1 << " (ID, x-coordinate, y-coordinate): ";
        cin >> house.id >> house.x >> house.y;
        houses.push_back(house);
    }

    cout << "Enter the pipeline connections (House1-ID House2-ID cost): " << endl;
    while (true) {
        string house1Id, house2Id;
        double cost;
        cin >> house1Id >> house2Id >> cost;
        if (house1Id == "-1") break;

        auto it1 = find_if(houses.begin(), houses.end(), [&](const House& h) { return h.id == house1Id; });
        auto it2 = find_if(houses.begin(), houses.end(), [&](const House& h) { return h.id == house2Id; });

        if (it1 != houses.end() && it2 != houses.end()) {
            it1->pipelines.push_back({house2Id, cost});
            it2->pipelines.push_back({house1Id, cost});
        } else {
            cout << "Invalid house ID(s). Try again." << endl;
        }
    }

    adjustPressure(houses);
    displayNetwork(houses);

    return 0;
}
